' ==============================================================================
' MACRO PRINCIPAL INTERACTIVA
' ==============================================================================
Sub InsertarImagenPersonalizada()
    Dim rutaImagen As String
    Dim colDestino As String, inputFila As String
    Dim filaBase As Long
    Dim fd As FileDialog
    
    Dim respBienvenida As VbMsgBoxResult
    Dim usarDesplazamiento As Boolean
    Dim respDesplazamiento As VbMsgBoxResult
    Dim respHojas As VbMsgBoxResult
    Dim hojaInicio As Long, hojaFin As Long
    Dim i As Long
    Dim totalHojas As Long

    ' 1. Validar que haya un libro abierto
    If ActiveWorkbook Is Nothing Then
        MsgBox "No hay ningún libro de Excel activo abierto.", vbExclamation
        Exit Sub
    End If

    ' 2. MENSAJE DE BIENVENIDA Y ADVERTENCIA
    respBienvenida = MsgBox("¡Bienvenido a la macro de inserción de imagen!" & vbCrLf & vbCrLf & _
                            "• Recuerde leer atentamente lo que se le solicita en cada paso." & vbCrLf & _
                            "• Se recomienda realizar un respaldo de este archivo en caso de errores." & vbCrLf & vbCrLf & _
                            "¿Desea continuar con la ejecución?", _
                            vbOKCancel + vbInformation, "Inicio - Inserción de Imagen")

    If respBienvenida = vbCancel Then Exit Sub

    totalHojas = ActiveWorkbook.Worksheets.Count

    ' 3. SELECCIONAR LA IMAGEN
    Set fd = Application.FileDialog(msoFileDialogFilePicker)
    fd.Title = "Selecciona la imagen a insertar"
    fd.Filters.Clear
    fd.Filters.Add "Imágenes", "*.png; *.jpg; *.jpeg"
    
    If fd.Show = -1 Then
        rutaImagen = fd.SelectedItems(1)
    Else
        Exit Sub ' El usuario canceló
    End If

    ' 4. UBICACIÓN (COLUMNA Y FILA BASE) - VALORES POR DEFECTO: "A" Y "1"
    colDestino = UCase(Trim(InputBox("Ingresa la COLUMNA donde colocarás la imagen (ej. A, F, Z):", "Ubicación", "A")))
    If colDestino = "" Then Exit Sub

    inputFila = InputBox("Ingresa el número de FILA donde deseas colocar la imagen:", "Ubicación", "1")
    If inputFila = "" Or Not IsNumeric(inputFila) Then Exit Sub
    filaBase = CLng(inputFila)

    ' 5. PREGUNTA DE MODO: ¿DESPLAZAR O COLOCAR EN FILA EXACTA?
    respDesplazamiento = MsgBox("¿Deseas activar el desplazamiento automático?" & vbCrLf & vbCrLf & _
                                "• SÍ = Si la fila tiene contenido/fondo, se moverá 5 filas abajo de lo último escrito." & vbCrLf & _
                                "• NO = Se colocará EXACTAMENTE en la fila indicada (" & filaBase & ").", _
                                vbYesNo + vbQuestion, "Modo de Inserción")
    
    usarDesplazamiento = (respDesplazamiento = vbYes)

    ' 6. PREGUNTA DE ALCANCE: ¿TODAS LAS HOJAS O UN RANGO? (BOTONES)
    respHojas = MsgBox("¿Deseas aplicar la imagen en TODAS las hojas del libro?" & vbCrLf & vbCrLf & _
                       "• SÍ = Aplicar a TODAS las hojas (1 a " & totalHojas & ")" & vbCrLf & _
                       "• NO = Especificar un RANGO de hojas (ej. de la 2 a la 5)", _
                       vbYesNoCancel + vbQuestion, "Alcance de Hojas")

    If respHojas = vbCancel Then Exit Sub

    If respHojas = vbYes Then
        ' Opción: TODAS las hojas
        hojaInicio = 1
        hojaFin = totalHojas
    Else
        ' Opción: RANGO específico de hojas
        inputFila = InputBox("Ingresa el número de la PRIMERA hoja (de 1 a " & totalHojas & "):", "Hoja Inicial", "1")
        If inputFila = "" Or Not IsNumeric(inputFila) Then Exit Sub
        hojaInicio = CLng(inputFila)

        inputFila = InputBox("Ingresa el número de la ÚLTIMA hoja (de " & hojaInicio & " a " & totalHojas & "):", "Hoja Final", CStr(totalHojas))
        If inputFila = "" Or Not IsNumeric(inputFila) Then Exit Sub
        hojaFin = CLng(inputFila)

        ' Validar límites del rango
        If hojaInicio < 1 Then hojaInicio = 1
        If hojaFin > totalHojas Then hojaFin = totalHojas
        If hojaInicio > hojaFin Then
            MsgBox "El rango introducido no es válido.", vbCritical
            Exit Sub
        End If
    End If

    ' 7. EJECUTAR INSERCIÓN EN EL RANGO SELECCIONADO
    For i = hojaInicio To hojaFin
        Call InsertarImagenInteligente(ActiveWorkbook.Worksheets(i), rutaImagen, colDestino, filaBase, usarDesplazamiento)
    Next i

    MsgBox "Imagen insertada con éxito en las hojas " & hojaInicio & " a la " & hojaFin & ".", vbInformation
End Sub


' ==============================================================================
' SUBRUTINA REUTILIZABLE
' ==============================================================================
Sub InsertarImagenInteligente(ws As Worksheet, rutaImg As String, columna As String, filaBase As Long, evaluarContenido As Boolean)
    Dim rngUsado As Range
    Dim iFila As Long, ultimaFila As Long
    Dim filaTieneAlgo As Boolean
    Dim celda As Range
    Dim filaFinal As Long
    Dim img As Picture

    ' Determinar la fila de destino según el modo elegido
    If evaluarContenido Then
        ' MODO DINÁMICO: Buscar la última fila con contenido o color de fondo
        Set rngUsado = ws.UsedRange
        ultimaFila = 0

        If Not rngUsado Is Nothing Then
            For iFila = (rngUsado.Row + rngUsado.Rows.Count - 1) To 1 Step -1
                filaTieneAlgo = False
                For Each celda In Intersect(ws.Rows(iFila), rngUsado)
                    If Trim(CStr(celda.Value)) <> "" Or celda.Interior.ColorIndex <> xlNone Then
                        filaTieneAlgo = True
                        Exit For
                    End If
                Next celda

                If filaTieneAlgo Then
                    ultimaFila = iFila
                    Exit For
                End If
            Next iFila
        End If

        If ultimaFila >= filaBase Then
            filaFinal = ultimaFila + 5
        Else
            filaFinal = filaBase
        End If
    Else
        ' MODO EXACTO: Colocar directamente en la fila solicitada
        filaFinal = filaBase
    End If

    ' Insertar imagen en la hoja dada
    Set img = ws.Pictures.Insert(rutaImg)
    With img
        .Top = ws.Range(columna & filaFinal).Top
        .Left = ws.Range(columna & filaFinal).Left
        .Width = 100
        .Height = 40
        .Placement = xlMoveAndSize
    End With
End Sub
